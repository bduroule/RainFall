class N {
	char *string;
	char string[108];
	int number;

	N(int number) {
		this->number = number;
	}

	int operator+(N &arg) {
		return this->number + arg.number;
	}

	void setAnnotation(char *str) {
		size_t len;

		len = strlen(str);

		memcpy(this->string, str, len);
		return;
	}
};

int main(int ac,char **av) {
	N nclass_1;
	N nclass_2;

	if (ac < 2) {
		exit(1);
	}

	nclass_1 = new N(5);

	nclass_2 = new N(6); 
	
	nclass_1->setAnnotation(av[1]);

	return nclass_2.operator+(&nclass_1);;
}
